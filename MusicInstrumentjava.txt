class MusicInstrument {

    int id;
    String name;
    String type;

    public MusicInstrument(int id, String name, String type) {
        this.id = id;
        this.name = name;
        this.type = type;
    }

    public void display() {
        System.out.println("Instrument: " + id + " " + name + " " + type);
    }
}
